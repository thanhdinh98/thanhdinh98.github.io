Link dbdiagram.io

SQL scripts:

CREATE TABLE IF NOT EXISTS game_portal.lucky_spin_item (
  id INT unsigned NOT NULL AUTO_INCREMENT,
  name VARCHAR(100) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL UNIQUE,
  currency CHAR(5) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  create_time bigint NOT NULL,

  PRIMARY KEY(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS game_portal.lucky_spin_item_set (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  description VARCHAR(100) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  lucky_spin_item_id INT UNSIGNED NOT NULL,
  threshold INT UNSIGNED NOT NULL DEFAULT 0,
  amount INT UNSIGNED NOT NULL,
  probability float UNSIGNED NOT NULL,
  remaining INT UNSIGNED NOT NULL DEFAULT 0,
  order_number INT UNSIGNED NOT NULL,
  set_id CHAR(32) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  create_time bigint NOT NULL,

  PRIMARY KEY(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS game_portal.lucky_spin_daily (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  lucky_spin_item_set_id char(32) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  status INT NOT NULL,
  create_time bigint NOT NULL,
  update_time bigint NOT NULL,

  PRIMARY KEY(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS game_portal.lucky_spin_daily_txn (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  subject varchar(44) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  network varchar(10) CHARACTER SET ascii COLLATE ascii_general_ci NOT NULL,
  amount INT UNSIGNED NOT NULL,
  lucky_spin_item_set_id INT UNSIGNED NOT NULL,
  create_time bigint NOT NULL,

  PRIMARY KEY(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

1 Adminâ€‹

1.1 Create lucky spin dailyâ€‹

Endpoint:â€‹

  POST /admin/lucky-spin/create/
  
  type (
    LuckySpinCreateRequest struct {
      Items []LuckySpinItem `json:"items" validate:"required"`
    }

    LuckySpinItem struct {
      Name        string         `json:"name" validate:"required"`
      Description string         `json:"description" validate:"required"`
      Currency    gmeta.Currency `json:"currency" validate:"required"`
      Amount      uint32         `json:"amount" validate:"required;min=0"`
      Threshold   uint32         `json:"threshold" validate:"required;min=0"`
      Probability float32        `json:"probability" validate:"required;min=0"`
    }

    LuckySpinCreateResponse struct {
      ID uint64 `json:"id"`
    }
  )

Validation:â€‹

  • Total probability should be 1
  • Item set need to have 1 default item (1 item with threshold = 0)

Test script:â€‹

  Scenario: Create daily spin with total probability over 100 percent
    Given a itemset with sequence percentage of "0.2,0.4,0.1,0.4"
    When create daily spin
    Then server return "error_invalid_data" code

  Scenario: Create daily spin with custom default item
    Given a itemset with "<number>" default item request
    When create daily spin
    Then server return "<status>" code

  # item with 0 threshold will set as default
  Examples:
      | number | status             |
      | 0      | error_invalid_data |
      | 1      | success            |
      | 2      | error_invalid_data |

  Scenario: Create daily spin successfully
    Given a correct itemset request
    When create daily spin
    Then server return "success" code

  Scenario: Create daily spin successfully with no duplicated spin item
    Given a correct itemset request
    When create daily spin
    Then server return "success" code
    Then create another daily spin with same itemset request
    And server return "success" code
    And item data is not duplicated in DB

1.2 Toggle lucky spin dailyâ€‹

Endpoint:â€‹

  POST /admin/lucky-spin/toggle/

  type (
    LuckySpinToggleRequest struct {
      LuckySpinDailyID uint64 `json:"id" validate:"required"`
    }

    LuckySpinToggleResponse struct {
      ID     uint64             `json:"id"`
      Status gmeta.CommonStatus `json:"status"`
    }
  )

Validation:â€‹

  • LuckySpinDailyID need to be exists

Test script:â€‹

  Scenario: Toggle lucky spin
    Given a correct itemset request
    Then create "<number>" daily spins with spin "<enable_spin>" enable
    And server return "success" code
    When toggle daily spin "<toggle_spin>"
    Then server return "<status>" code
    And spin "<toggle_spin>" is "<current_spin_status>" and others are "<other_spin_status>"
  
  # spin status:
  # 1: enable
  # -1: disable
  Examples:
      | number | enable_spin | toggle_spin | current_spin_status | other_spin_status | status               |
      | 2      | 1           | 2           | enable              | disable           | success              |
      | 2      | 1           | 1           | disable             | disable           | success              |
      | 2      | 0           | 1           | enable              | disable           | success              |
      | 2      | 1           | 3           | not_found           | not_found         | error_data_not_found |
